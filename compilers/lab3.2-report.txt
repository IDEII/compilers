%start NProgram;

NProgram:
NProgram NFunction {$$ = $1 + '\n' + $2; driver.result = $$;}
| NFunction {$$ = $1;}
;


NFunction:
	NFunctionHeader TAG_FUNCBODYSTART inc NFuncBody dec {
	$$ = $1 + ' ' + $2;
	$$ += '\n' + std::string(driver.enclosure + 1, '\t');
	$$ += $4;
	$$ += '\n' + std::string(driver.enclosure, '\t');
}
;

inc: %empty {driver.enclosure++;};
dec: %empty {driver.enclosure--;};

NFunctionHeader:
	NType  TAG_VARNAME TAG_BACK NFuncParamStatements {$$ = $1 + ' ' + $2 + ' ' + $3 + ' ' + $4;}
	| TAG_VOID TAG_VARNAME TAG_BACK NFuncParamStatements {$$ = $1 + ' ' + $2 + ' ' + $3 + $4;}
	| TAG_VOID TAG_VARNAME {$$ = $1 + ' ' + $2;}
;


NType:
	NBasicType {$$ = $1;}
	| NType TAG_LBC TAG_RBC {$$ = $1 + $2 + $3;} 
;

NBasicType: 
	TAG_INT {$$ = $1;}
	| TAG_CHAR {$$ = $1;}
	| TAG_BOOL {$$ = $1;};

NFuncParamStatements: 
	NFuncParams {$$ = $1;}
;

NFuncParams: 
	NFuncParams TAG_COMMA NFuncParam {$$ = $1 + ' ' + $2 + ' ' + $3;}
	| NFuncParam {$$ = $1;}
;

NFuncParam: 
	NType TAG_VARNAME {$$ = $1 + ' ' + $2;}
;

NFuncBody:
	NStatements  TAG_DOT {$$ = $1 + '\n' + std::string(driver.enclosure, '\t') + $2;} 
;

NStatements:
	NStatements TAG_SEMICOLON NStatement {$$ = $1 + $2 + '\n' + std::string(driver.enclosure, '\t') + $3;}
	| NStatement {$$ = $1;}
;


NStatement:
	NDeclarationStatement {$$ = $1;}
	| NAssignmentStatement {$$ = $1;}
	| NCallExpression {$$ = $1;}
	| NIfStatement {$$ = $1;}
	| NIfElseStatement {$$ = $1;}
	| NLoopWhileStatement {$$ = $1;}
	| NLoopForStatement {$$ = $1;}
	| NReturnStatement {$$ = $1;}
	| NLoopPostStatement {$$ = $1;}
;

NDeclarationStatement:
	NType NVarDeclarations {$$ = $1 + ' ' + $2;}
;

NVarDeclarations: 
	NVarDeclarations TAG_COMMA NVarDeclaration {$$ = $1 + $2 + ' ' + $3;}
	| NVarDeclaration {$$ = $1;}
;

NVarDeclaration: 
	TAG_VARNAME TAG_ASSIGN NArithmeticExpression {$$ = $1 + ' ' + $2 + ' ' + $3;}
	| TAG_VARNAME {$$ = $1;}
;

NAssignmentStatement: 
	NSpec TAG_ASSIGN NExpr {$$ = $1 + ' ' + $2 + ' ' + $3;}
;

NSpec: 
	TAG_LB NType NPrim TAG_RB {$$ = $1 + $2 + ' ' + $3 + $4;}
	| NSpec NPrim {$$ = $1 + ' ' + $2;}
	| NPrim {$$ = $1;}
;
	
NPrim:
	NConst {$$ = $1;}
	| TAG_VARNAME {$$ = $1;}
	| TAG_LB NExpr TAG_RB {$$ = $1 + ' ' + $2 + ' ' + $3;}
;


NConst: 
	TAG_NUMBER {$$ = $1;}
	| TAG_CHAR_CONST {$$ = $1;}
	| TAG_REF_CONST {$$ = $1;}
	| TAG_STRING_CONST {$$ = $1;}
	| TAG_TRUE {$$ = $1;}
	| TAG_FALSE {$$ = $1;}
;

NExpr: 
	NLogicalExpression {$$ = $1;}
	| NExpr TAG_OR NLogicalExpression {$$ = $1 + ' ' + $2 + ' ' + $3;}
	| NExpr TAG_XOR NLogicalExpression {$$ = $1 + ' ' + $2 + ' ' + $3;}
;

NLogicalExpression: 
	NCompareExpression {$$ = $1;}
	| NLogicalExpression TAG_AND NCompareExpression {$$ = $1 + ' ' + $2 + ' ' + $3;}
;

NCompareOperator:
	TAG_EQ {$$ = $1;}
	| TAG_NE {$$ = $1;}
	| TAG_LE {$$ = $1;}
	| TAG_LT {$$ = $1;}
	| TAG_GE {$$ = $1;}
	| TAG_GT {$$ = $1;}
;


NCompareExpression: 
	NCallExpression {$$ = $1;}
	| NCompareExpression NCompareOperator NCallExpression {$$ = $1 + ' ' + $2 + ' ' + $3;}
;

NCallExpression: 
	NArithmeticExpression {$$ = $1;}
	| TAG_VARNAME TAG_BACK NArgs {$$ = $1 + ' ' + $2 + ' ' + $3;}
;

NArgs: 
	NArithmeticExpression {$$ = $1;}
	| NArgs TAG_COMMA NArithmeticExpression {$$ = $1 + ' ' + $2 + ' ' + $3;}
;

NAdditiveOperator:
	TAG_PLUS {$$ = $1;}
	| TAG_MINUS {$$ = $1;}
;

NArithmeticExpression: 
	NMultiplicativeExpression {$$ = $1;}
	| NArithmeticExpression NAdditiveOperator NMultiplicativeExpression {$$ = $1 + ' ' + $2 + ' ' + $3;}
;

NMultiplicativeExpression: 
	NTerm {$$ = $1;}
	| NMultiplicativeExpression NMultiplicativeOperator NTerm {$$ = $1 + ' ' + $2 + ' ' + $3;}
;

NMultiplicativeOperator: 
	TAG_MUL {$$ = $1;}
	| TAG_DIV {$$ = $1;}
	| TAG_MOD {$$ = $1;}
;

NTerm:  
	NFactor {$$ = $1;}
	| NFactor TAG_POW NTerm {$$ = $1 + ' ' + $2 + ' ' + $3;}
;

NFactor:
	TAG_NOT NSpec {$$ = $1 + ' ' + $2;}
	| TAG_MINUS NSpec {$$ = $1 + ' ' + $2;}
	| NSpec {$$ = $1;}
; 

NIfElseStatement: 
	NExpr TAG_THEN inc NStatements dec NElseBlock TAG_DOT {
	$$ = $1 + ' ' + $2;
	$$ += '\n' + std::string(driver.enclosure + 1, '\t');
	$$ += $4;
	$$ += '\n' + std::string(driver.enclosure, '\t');
	$$ += $6 + ' ' + $7;
}
;

NIfStatement: 
	NExpr TAG_THEN inc NStatements TAG_DOT dec {
	$$ = $1 + ' ' + $2;
	$$ += '\n' + std::string(driver.enclosure + 1, '\t');
	$$ += $4 + ' ' + $5;
}
;

NElseBlock: 
	TAG_ELSE inc NStatements dec {
	$$ = $1;
	$$ += '\n' + std::string(driver.enclosure + 1, '\t');
	$$ += $3;
}
;


NLoopWhileStatement: 
	NExpr TAG_LOOP inc NStatements TAG_DOT dec {
	$$ = $1 + ' ' + $2;
	$$ += '\n' + std::string(driver.enclosure + 1, '\t');
	$$ += $4 + ' ' + $5;
} 
;

NLoopForStatement: 
	NCycle inc NStatements TAG_DOT dec {
	$$ = $1;
	$$ += '\n' + std::string(driver.enclosure + 1, '\t');
	$$ += $3 + ' ' + $4;
} 
;

NLoopPostStatement: 
	TAG_LOOP inc NStatements dec TAG_WHILE NExpr TAG_DOT {
	$$ = $1;
	$$ += '\n' + std::string(driver.enclosure + 1, '\t');
	$$ += $3;
	$$ += '\n' + std::string(driver.enclosure, '\t');
	$$ += $5 + ' ' + $6 + ' ' + $7;
}
;

NCycle: 
	NExpr TAG_TILDA NExpr TAG_LOOP NCycleVar {$$ = $1 + ' ' + $2 + ' ' + $3 + ' ' + $4 + ' ' + $5;}
;
NCycleVar: 
	TAG_VARNAME {$$ = $1;}
;

NReturnStatement: 
	TAG_RETURN NExpr {$$ = $1 + ' ' + $2;}
;

# Вывод
В ходе лабораторной работы научился применять генератор синтаксических анализаторов bison. Лабораторная
работа не была сложной, потому что уже был опыт работы с flex из лабораторной работы 1.5, а так же потому
что на момент написания этой лабораторной работы у меня, наконец, появилась грамматика для языка L2.
Поэтому лабораторная работа по большей части состояла из переписывания грамматики в форму bison.
Лабораторная была бы труднее, если бы я ее делал не на C++, а на C, так как последний не позволяет
создавать удобные конструкции для форматирования, как это представлено у меня. Так же во время
тестирования программы я столкнулся с проблемой, связанной с ошибками компилятора GCC, когда для того,
чтобы собрать проект подключались библиотеки не имеющие никакого отношения к программе, и поэтому я
сменил компилятор на CLang.
