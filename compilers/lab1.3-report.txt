# Цель работы
Целью данной работы является приобретение навыка реализации лексического анализатора на
объектно-ориентированном языке без применения каких-либо средств автоматизации решения задачи
лексического анализа.

# Индивидуальный вариант
-   Химические вещества: последовательности латинских букв и цифр, начинающиеся с заглавной буквы, при
этом после цифры не может следовать строчная буква (атрибут: строка). Примеры: «`CuSO4`», «`CH3CH2OH`»,
«`Fe2O3`».
-   Коэффициенты: последовательности десятичных цифр. Между коэффициентом и веществом пробел может
отсутствовать.
-   Операторы: «`+`», «`->`».

##### класс `Position`:

```python
 class Position:  
    def __init__(self, program):  
        self.__line = 1  
  self.__pos = 1  
  self.__index = 0  
  self.__program = program  
  
    def get_char(self):  
        return self.__program[self.__index]  
  
    def __str__(self):  
        return f'({self.__line}, {self.__pos})'  
  
  def is_new_line(self):  
        if self.__index == '\r' and self.__index + 1 < len(self.__program):  
            return self.__program[self.__index + 1] == '\n'  
  return self.__program[self.__index] == '\n'  
  
  def is_digit(self):  
        return self.__program[self.__index].isdigit()  
  
    def is_letter(self):  
        return self.__program[self.__index].isalpha()  
  
    def is_lower(self):  
        return self.__program[self.__index].islower()  
  
    def is_space(self):  
        return self.__program[self.__index].isspace()  
  
    def is_spectial_simbol(self):  
        if self.__program[self.__index] == "+":  
            return True  
  
 if self.__program[self.__index] == "-":  
            return True  
  
 if self.__program[self.__index] == ">":  
            return True  
 return False  
 def next(self):  
        if self.__index != len(self.__program):  
            if self.is_new_line():  
                if self.__program[self.__index] == '\r':  
                    self.__index += 1  
  self.__line += 1  
  self.__pos = 1  
  else:  
                self.__pos += 1  
  self.__index += 1  
  
  def raise_error(self):  
        print("Не удалось распознать лексему: " + self.__str__())  
        while not self.is_space() and not self.is_new_line():  
            self.next()  
            if self.is_new_line():  
                self.next()  
        self.skip_spaces()  
  
    def skip_spaces(self):  
        while self.is_space():  
            self.next()  
  
  
```

##### класс `Fragment`:

```python 
class Fragment:  
    def __init__(self, start, follow):  
        self.__start = start  
        self.__follow = follow  
  
    def __str__(self):  
        return self.__start.__str__() + "-" + self.__follow.__str__()  
  
```

##### класс `DomainTag`:

```python
class DomainTag:  
	ELEMENT = 0  
	COEFF = 1  
	OPERATION = 2  
	END_OF_PROGRAM = 3  
```

##### классы `Token`, `ErrorToken`,`OperationToken`,`CoeffToken`, `ElementToken`, `EndToken`:

```python
class Token:  
    def __init__(self, tag, start, follow):  
        self.tag = tag  
        self.coords = Fragment(start, follow)  
        super().__init__()  
  
class ErrorToken(Token):  
    def __init__(self, code, start, follow):  
        super(ErrorToken, self).__init__(DomainTag.COEFF, start, follow)  
        self.code = code  
  
    def __str__(self):  
        return 'ERROR ' + self.coords.__str__() + ": " + str(self.code)  
  
class OperationToken(Token):  
    def __init__(self, code, start, follow):  
        super(OperationToken, self).__init__(DomainTag.OPERATION, start, follow)  
        self.code = code  
  
    def __str__(self):  
        return 'OPERATION ' + self.coords.__str__() + ": " + str(self.code)  
  
class CoeffToken(Token):  
    def __init__(self, code, start, follow):  
        super(CoeffToken, self).__init__(DomainTag.COEFF, start, follow)  
        self.code = code  
  
    def __str__(self):  
        return 'COEFF ' + self.coords.__str__() + ": " + str(self.code)  
  
  
class ElementToken(Token):  
    def __init__(self, code, start, follow):  
        super(ElementToken, self).__init__(DomainTag.ELEMENT, start, follow)  
        self.code = code  
  
    def __str__(self):  
        return 'ELEMENT ' + self.coords.__str__() + ": " + str(self.code)  
  
class EndToken(Token):  
    def __init__(self, start, follow):  
        super(EndToken, self).__init__(DomainTag.END_OF_PROGRAM, start, follow)  
  
    def __str__(self):  
        return 'END ' + self.coords.__str__()  
```
