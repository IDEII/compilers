# Индивидуальное задание
Язык L2.
Семантический анализ:
- Вызываемые функции должны быть определены в программе.
- Переменные в выражениях должны быть определены ранее.
- Область видимости переменной начинается с оператора-объявления и заканчивается 
вместе с последовательностью операторов, где она определена.
- Операции применяются к операндам допустимых типов (таблица 3).
- Операнды оператора присваивания должны быть совместимых типов.
- Типы формальных параметров должны совпадать с типами фактических параметров.

```python
@dataclass  
class BinOperatorExpression(Expr):  
    left: Expr  
    operator: str  
  operator_coord: pe.Position  
    right: Expr  
    type: BasicType or Type  
  
    @pe.ExAction  
  def create(attrs, coords, res_coord):  
        left, operator, right = attrs  
  return BinOperatorExpression(left, operator, coords[1].start, right, None)  
  
    def check(self, funcs, vars):  
        self.left.check(funcs, vars)  
        self.right.check(funcs, vars)  
  
        is_int = lambda a: a in (BasicType.Integer, BasicType.Char)  
        lt = self.left.type  
        rt = self.right.type  
        if self.operator in ('<', '<=', '>', '>='):  
            if is_int(lt) and is_int(rt):  
                self.type = BasicType.Boolean  
            else:  
                raise BinaryOperatorTypeMismatch(self.operator_coord, self.left.type, self.operator,
self.right.type)  
        elif self.operator == '+' and (is_int(lt) or is_int(rt)):  
            if lt == BasicType.Integer and rt == BasicType.Integer:  
                self.type = BasicType.Integer  
            else:  
                self.type = BasicType.Char  
        elif self.operator in ('|', '&', '^'):  
            if lt == BasicType.Boolean and rt == BasicType.Boolean:  
                self.type = BasicType.Boolean  
            else:  
                raise BinaryOperatorTypeMismatch(self.operator_coord, self.left.type, self.operator,
self.right.type)  
        elif self.operator in ('==', '!='):  
            if lt != String and rt != String:  
                if is_int(lt) and is_int(rt):  
                    self.type = BasicType.Boolean  
                elif lt == rt:  
                    self.type = BasicType.Boolean  
                elif (type(lt) == ArrayType or isinstance(self.left, RefConst)) and \  
                        (type(rt) == ArrayType or isinstance(self.right, RefConst)):  
                    self.type = BasicType.Boolean  
        elif self.operator in ('*', '/', '%', '^'):  
            if lt == BasicType.Integer and rt == BasicType.Integer:  
                self.type = BasicType.Integer  
            else:  
                raise BinaryOperatorTypeMismatch(self.operator_coord, self.left.type, self.operator,
self.right.type)  
        elif self.operator == '-':  
            if is_int(lt) and is_int(rt):  
                if lt == BasicType.Char and rt == BasicType.Integer:  
                    self.type = BasicType.Char  
                else:  
                    self.type = BasicType.Integer  
        else:  
            raise BinaryOperatorTypeMismatch(self.operator_coord, self.left.type, self.operator,
self.right.type)

@dataclass  
class AssignmentStatement(Statement):  
    left: Any  
    right: Any  
    coord: pe.Position  
  
    def check(self, funcs, vars):  
        print('Assignment Statement')  
        pprint(self.left)  
        pprint(self.right)  
        self.right.check(funcs, vars)  
        if type(self.left) not in (Var, ArrayCall):  
            if funcs[self.left].return_type != self.right.type:  
                raise AssignTypeMismatch(self.coord, funcs[self.left].return_type, self.right.type)  
        elif type(self.left) == Var and self.left.name in vars:  
            self.left = Var(self.left, self.coord, vars[self.left.name])  
        elif type(self.left) == ArrayCall:  
            self.left.check(funcs, vars)  
            self.right.check(funcs, vars)  
            if self.left.type != self.right.type:  
                raise AssignTypeMismatch(self.coord, self.left.type, self.right.type)  
        elif type(self.left) == Var:  
            if self.left.name not in funcs and self.left.name not in vars:  
                raise Exception('Unknown reference')  
        else:  
            self.left.check(funcs, vars)  
            self.right.check(funcs, vars)  
            if self.left.type != self.right.type:  
                raise AssignTypeMismatch(self.coord, self.left.type, self.right.type)  
  
  
    @pe.ExAction  
  def create(attrs, coords, res_coord):  
        left, right = attrs  
  return AssignmentStatement(left, right, coords[1])  
  
@dataclass  
class FunctionCallStatement(Statement):  
    name: Any  
    args: list[Expr]  
    name_coord: pe.Position  
    args_coord: pe.Position  
    type: Type or BasicType  
  
    def check(self, funcs, vars):  
        print(f'checking function call{self.name}')  
        if self.name not in funcs:  
            raise UndefinedFunction(self.name_coord, self.name)  
        if len(self.args) != funcs[self.name].args_len():  
            raise ArgumentsLenMismatch(self.args_coord, self.name, funcs[self.name].args_len(),
len(self.args))  
  
        self.type = funcs[self.name].return_type  
        params_map_types = funcs[self.name].params_map_types()  
        for arg in self.args:  
            arg.check(funcs, vars)  
  
        for param, arg in zip(params_map_types, self.args):  
            if param[1] != arg.type:  
                raise ArgumentsTypeMismatch(self.args_coord, self.name, param[1].type, arg.type)  
  
  
    @pe.ExAction  
  def create(attrs, coords, res_coord):  
        name, args = attrs  
  return FunctionCallStatement(name, args, coords[0].start, coords[1].start, None)

@dataclass  
class IfStatement(Statement):  
    condition: Expr  
    statements: list[Statement]  
    elseblock: list[Statement]  
  
    def check(self, funcs, vars):  
        self.condition.check(funcs, vars)  
        inner_vars = copy.deepcopy(vars)  
        for stmt in self.statements:  
            stmt.check(funcs,  inner_vars)  
        inner_vars_else = copy.deepcopy(vars)  
        for stmt in self.elseblock:  
           stmt.check(funcs, inner_vars_else)

@dataclass  
class CycleVar:  
    name: Any  
    value: Any  
    type: Optional[Type]  
  
    def check(self, funcs, vars):  
        if self.name in vars:  
            raise AlreadyExists(None, self.name)  
        self.value.check(funcs, vars)  
        if self.type is None:  
            self.type = self.value.type  
        vars[self.name] = self.type  
  
  
@dataclass  
class ForCycleHead:  
    for_var: CycleVar  
    forfrom: Any  
    to: Any  
  
    def check(self, funcs, vars):  
        # pprint(vars)  
  self.for_var.value = self.forfrom  
        self.for_var.check(funcs, vars)  
        self.forfrom.check(funcs, vars)  
        self.to.check(funcs, vars)  
        if self.for_var.type != self.to.type:  
            raise TypeMismatch(None, self.for_var.type, self.to.type)

class Function:  
    header: FunctionHeaderFull or FunctionHeaderShort  
    body: Body  
  
    def check(self, funcs):  
        vars = {}  
        if isinstance(self.header, FunctionHeaderFull):  
            params = self.header.params  
            for param in params:  
                param_name1 = param.param_name  
                param_type1 = param.param_type  
                if param_name1 in vars and param_type1 == vars[param_name1]:  
                    raise AlreadyExists(param.name_pos, param_name1)  
                vars[param_name1] = param_type1  
            vars[self.header.name] = self.header.return_type  
            print(f'\nchecking function {self.header.name}')  
            pprint(vars)  
            self.body.check(funcs, self.header.return_type, vars)  
  
  
@dataclass  
class Program:  
    entities: list[Function]  
  
    def check(self):  
        funcs = {}  
        for func in self.entities:  
            print(func.header.name)  
            if func.header.name in funcs:  
                raise AlreadyExists(func.header.name_pos, func.header.name)  
            funcs[func.header.name] = func.header  
  
        pprint(funcs)  
        for func in self.entities:  
            vars = {}  
            func.check(funcs)
```
