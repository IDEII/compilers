NProgram -> NFunction 
| NProgram NFunction .

NFunction -> NFunctionHeader "=" NFuncBody .

NFunctionHeader -> NType '{' VARNAME '}' "<-" NFuncParamStatements .
NFunctionHeader -> "void" '{' VARNAME '}' "<-" NFuncParamStatements .
NFunctionHeader -> "void" '{' VARNAME '}' .



NType -> NBasicType | NType "[]" .
NBasicType -> KW_INT | KW_CHAR | KW_BOOL .

NFuncBody -> NStatements '.' .
NFuncParamStatements -> NFuncParams .
NFuncParams -> NFuncParams ',' NFuncParam 
| NFuncParam .

NFuncParam -> NType VARNAME .

NStatements -> NStatements ';' NStatement 
| NStatement .

NStatement -> NDeclarationStatement 
| NAssignmentStatement 
| NArrayAssignmentStatement 
| NCallExpression 
| NIfStatement 
| NIfElseStatement 
| NLoopWhileStatement 
| NLoopForStatement 
| NReturnStatement 
| NLoopPostStatement .

NDeclarationStatement -> NType NVarDeclarations .

NVarDeclarations -> NVarDeclarations ',' NVarDeclaration 
| NVarDeclaration .
NVarDeclaration -> '{' VARNAME '}'':=' NArithmeticExpression  ## нет возможности вызвать функцию без скобок '(' Func ')'
| '{' VARNAME '}'.

## NAssignmentStatement -> '{' VARNAME '}' ':=' NExpr .
NArrayAssignmentStatement -> NSpec ':=' NExpr .

NSpec -> NType NPrim
| NSpec NPrim 
| NPrim .

NPrim -> NConst | VARNAME | '(' NExpr ')' . ## '{' '}'

NConst -> INTEGER | CHAR | REF_CONST | STRING | KW_TRUE | KW_FALSE .

NExpr ->  NLogicalExpression 
| NExpr '|' NLogicalEXpression 
| NExpr '@' NLogicalEXpression .

NLogicalExpression -> NCompareExpression 
| NLogicalExpression '&' NCompareExpression .

NCompareOperator -> '==' | '!=' | '<' | '>' | '<=' | '=>' .

NCompareExpression -> NCallExpression
| NCompareExpression NCompareOperator NCallExpression .


NCallExpression -> NArithmeticExpression 
| '{' VARNAME '}' '<-' NArgs .
NArgs -> NArithmeticExpression | NArgs ',' NArithmeticExpression .

NAdditiveOperator -> '+' | '-' .

NArithmeticExpression -> NMultiplicativeExperssion 
| NArithmeticExpression NAdditiveOperator NMultiplicativeExperssion .


NMultiplicativeExperssion -> NTerm 
| NMultiplicativeExperssion NMultiplicativeOperator NTerm .

NMultiplicativeOperator -> '*' | '/' | '%' .

NTerm -> NFactor | NFactor '^' NTerm .

NFactor -> '!' NSpec | '-' NSpec | NSpec .

NIfElseStatement -> NExpr 'then' NStatements NElseBlock '.' .

NIfStatement -> NExpr 'then' NStatements '.' .

NElseBlock -> 'else' NStatements .

NLoopWhileStatement -> NExpr 'loop' NStatements '.' .
NLoopForStatement -> NCycle NStatements '.' .
NLoopPostStatement -> 'loop' NStatements 'while' NExpr '.' .

NCycle -> NExpr '~' NExpr 'loop' NCycleVar .
NCycleVar -> '{' VARNAME '}'.

NReturnStatement -> 'return' NExpr .




INTEGER -> '1111111' . 
CHAR -> 'Z' .
KW_INT -> 'int' .
KW_CHAR -> 'char' .
KW_BOOL -> 'bool' .
VARNAME -> 'varname' .
KW_TRUE -> 'true' .
KW_FALSE -> 'false' .
REF_CONST -> 'null' .
