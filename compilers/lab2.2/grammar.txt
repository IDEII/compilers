Program -> Program Func | Func
Func ->  FuncHeader FuncBody
FuncHeader -> Type VARNAME <- FuncParams | void VARNAME <- FuncParams | Type VARNAME | void VARNAME
FuncParams -> FuncParams , BasicVar | BasicVar
BasicVar -> Type VARNAME
Type -> BasicType | Type []
BasicType -> INTEGER | CHAR | BOOL

PointerType -> BasicType | BasicType []

FuncBody -> = Statements .

Statements -> Statement | Statements ; Statement

Statement -> ExtendedVar := Expr
          | VARNAME <- Args
	  | Anons
          | BasicVar
          | Expr then Statements else Statements .
          | Expr then Statements .
          | Expr loop Statements .
          | Expr ~ Expr loop CycleVar Statements .
	  | loop Statements while expr .
          | return Expr


Args -> VARNAME | Args, VARNAME
CycleVar -> Type VARNAME

Expr -> LogicalExpr
     | Expr | LogicalExpr
     | Expr @ LogicalExpr

LogicalExpr -> CompareExpr | LogicalExpr & ComareExpr

CompareExpr -> ArithmExpr | ArithmExpr CmpOp ArithmExpr
CmpOp -> == | != | < | > | <= | >=

ArithmExpr -> PowExpr | ArithmExpr AddOp PowExpr
AddOp -> + | -

PowExpr -> Term | Term ^ PowExpr

Term -> Factor | Term MulOp Factor
MulOp -> * | / | %

Factor -> ! Spec | - Spec | Spec

Spec -> VARNAME <- Args | Type VARNAME | Type INT_CONST | Const | Var

ExtendedVar -> BasicVar | Var
Var -> (Spec)(Expr) | VARNAME
Var -> VARNAME INT_CONST

Anons -> Anons , Anon | Anon
Anon -> Vars := Expr
Vars -> Vars BasicVar | BasicVar

Const -> INT_CONST | CHAR_CONST | STRING_CONST | REF_CONST | TRUE | FALSE

VARNAME = \{[Unicode]*\}
FUNCNAME = [Unicode]
REF_CONST = null
INT_CONST  = ([A-Za-z0-9]+)\$\d|\d+)
CHAR_CONST  = $\"[Unicode]?\"
STRING_CONST  = \"[Unicode]*\"
